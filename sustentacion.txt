#------------PUNTO 9------------------------

var factorial = lambda(n) => {
    if (n <= 1) {
        1
    } else {
        var resultado = 1;
        while (n > 0){
            resultado = (resultado * n)
            n = (n - 1)
        }
        resultado
    }
};

var registro = {};

var registroFactorial = lambda(L) => {
    var listaFactoriales = [];
    for i in L {
        listaFactoriales = (listaFactoriales append ['factorial(i)])
    }
    registro.["valores"] = L
    registro.["factoriales"] = listaFactoriales
};

'registroFactorial([1,2,3,4,7,9])

print(registro)

print("valores:", registro.["valores"])
print("factoriales:", registro.["factoriales"])

return

  }else{

  }
};
